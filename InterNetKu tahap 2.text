/* ---------------- GLOBAL FUNCTION FOR STAGE CLICK INTERACTION ---------------- */
let selectStageDirectly; 

(function(){
  const path = document.getElementById('mainPath');
  const packet = document.getElementById('packet');
  const totalLen = path.getTotalLength();
  const stations = [0,1,2,3,4].map(i => document.getElementById('st'+i));
  const rings = stations.map(s => s.querySelector('.station-ring'));
  const labels = stations.map(s => s.querySelector('.station-label'));

  const stageTag = document.getElementById('stageTag');
  const stepNum = document.getElementById('stepNum');
  const stepTitle = document.getElementById('stepTitle');
  const stepDesc = document.getElementById('stepDesc');
  const playBtn = document.getElementById('playBtn');
  const resetBtn = document.getElementById('resetBtn');

  const stages = [
    {title:"Tahap 1 — Sinyal Digital", desc:"Laptop Sinta mengubah pesannya menjadi sinyal digital (0 dan 1) yang siap dikirim.", frac:0.06},
    {title:"Tahap 2 — Lewat Router", desc:"Sinyal berjalan melalui router/modem di rumah, keluar menuju jaringan luar.", frac:0.30},
    {title:"Tahap 3 — Masuk ISP", desc:"Data melewati ISP menuju jaringan internet global — jalan raya data dunia.", frac:0.52},
    {title:"Tahap 4 — Diproses Server", desc:"Data akhirnya tiba di server tujuan dan diproses sesuai permintaan.", frac:0.76},
    {title:"Tahap 5 — Respons Kembali", desc:"Server mengirim balik jawaban lewat jalur yang sama, sampai muncul di layar Sinta.", frac:1.0}
  ];

  let playing = false, rafId = null, startTime = null;
  const DURATION = 6200; 

  function setStationActive(index){
    rings.forEach((r,i)=> r.classList.toggle('active', i<=index));
    labels.forEach((l,i)=> l.classList.toggle('active', i<=index));
  }

  function placePacketAtStart(){
    const p = path.getPointAtLength(0);
    packet.setAttribute('cx', p.x);
    packet.setAttribute('cy', p.y);
  }

  function update(ts){
    if(!startTime) startTime = ts;
    const elapsed = ts - startTime;
    let t = Math.min(elapsed / DURATION, 1);
    const len = t * totalLen;
    const p = path.getPointAtLength(len);
    packet.setAttribute('cx', p.x);
    packet.setAttribute('cy', p.y);

    let idx = 0;
    for(let i=0;i<stages.length;i++){ if(t >= stages[i].frac - 0.001) idx = i; }
    setStationActive(idx);
    stageTag.textContent = "Sedang berjalan · " + stages[idx].title.split("—")[0].trim();
    stepNum.textContent = (idx+1);
    stepTitle.textContent = stages[idx].title;
    stepDesc.textContent = stages[idx].desc;

    if(t < 1){
      rafId = requestAnimationFrame(update);
    } else {
      playing = false;
      playBtn.textContent = "▶ Mulai";
      stageTag.textContent = "Selesai — data sudah kembali!";
    }
  }

  function play(){
    if(playing) return;
    playing = true;
    startTime = null;
    playBtn.textContent = "⏸ Berjalan…";
    rafId = requestAnimationFrame(update);
  }

  function reset(){
    playing = false;
    if(rafId) cancelAnimationFrame(rafId);
    startTime = null;
    placePacketAtStart();
    setStationActive(-1);
    stageTag.textContent = "Siap dimulai";
    stepNum.textContent = "0";
    stepTitle.textContent = "Belum dimulai";
    stepDesc.textContent = 'Tekan tombol "Mulai" di atas untuk melihat data berangkat dari perangkatmu.';
    playBtn.textContent = "▶ Mulai";
  }

  // Feature: Direct Stage Selection (Triggered on Node/Card Click)
  selectStageDirectly = function(idx) {
    if(playing) reset();
    
    // Teleport packet directly to selected node path
    const targetFrac = stages[idx].frac;
    const len = targetFrac * totalLen;
    const p = path.getPointAtLength(len);
    packet.setAttribute('cx', p.x);
    packet.setAttribute('cy', p.y);
    
    setStationActive(idx);
    stageTag.textContent = "Navigasi Manual · " + stages[idx].title.split("—")[0].trim();
    stepNum.textContent = (idx+1);
    stepTitle.textContent = stages[idx].title;
    stepDesc.textContent = stages[idx].desc + " (Dipilih secara manual via klik).";
  }

  // Bind click elements inside the interactive SVG nodes
  stations.forEach((station, index) => {
    station.addEventListener('click', () => {
      selectStageDirectly(index);
      document.getElementById('simulasi').scrollIntoView();
    });
  });

  playBtn.addEventListener('click', play);
  resetBtn.addEventListener('click', reset);
  placePacketAtStart();
})();

(function(){
  const loginScreen = document.getElementById('loginScreen');
  const mainContent = document.getElementById('mainContent');
  const loginBtn = document.getElementById('loginBtn');
  const logoutBtn = document.getElementById('logoutBtn');
  const usernameInput = document.getElementById('username');
  const passwordInput = document.getElementById('password');
  const loginError = document.getElementById('loginError');

  // Check auth state from storage
  if (sessionStorage.getItem('internetku_logged') === 'true') {
    loginScreen.style.display = 'none';
    mainContent.style.display = 'block';
  }

  function handleLogin() {
    const user = usernameInput.value.trim();
    const pass = passwordInput.value;

    // Hardcoded credentials logic
    if (user === 'siswa' && pass === '123456') {
      loginError.style.display = 'none';
      sessionStorage.setItem('internetku_logged', 'true');
      
      // Dynamic transitional effect
      loginScreen.style.transition = 'opacity 0.4s ease';
      loginScreen.style.opacity = '0';
      setTimeout(() => {
        loginScreen.style.display = 'none';
        mainContent.style.display = 'block';
      }, 400);
    } else {
      loginError.style.display = 'block';
      // Mini shake animation effect on mistake
      usernameInput.style.borderColor = 'var(--coral)';
      passwordInput.style.borderColor = 'var(--coral)';
    }
  }

  loginBtn.addEventListener('click', handleLogin);
  passwordInput.addEventListener('keypress', (e) => { if (e.key === 'Enter') handleLogin(); });

  logoutBtn.addEventListener('click', () => {
    sessionStorage.removeItem('internetku_logged');
    window.location.reload();
  });
})();
